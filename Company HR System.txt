class Employee {
protected:
    string name;
    int id ;
    double sallery;
public:

        this->name=name;this ->id=id;this->sallery=sallery;
    }
    virtual string get_Role() {
        return  "Employee";
    }
    virtual string get_name() {
        return name;
    }
    virtual int get_id() {
        return id;
    }
    virtual double get_sallery() {
        return sallery;
    }
};
